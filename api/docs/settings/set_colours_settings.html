appointment_types_attributes Array of appointment types
appointment_types_attributes[appointment_type_id] Appointment type id

appointment_types_attributes[_destroy] Boolean, if true - delete
appointment_statuses_attributes Array of appointment statuses

appointment_statuses_attributes[colour] Appointment status color
